====== Example ======
<code clipper>
#include "FiveWin.ch"
static oWnd

//----------------------------------------------------------------------------//
function Main()

   DEFINE WINDOW oWnd TITLE "Testing Window" ;
      MENU BuildMenu()

   ACTIVATE WINDOW oWnd

return nil

function BuildMenu()

   local oMenu

   MENU oMenu
      MENUITEM "Call a Dialog" ACTION TestDlg()
   ENDMENU

return oMenu

//----------------------------------------------------------------------------//

function TestDlg()

   local oDlg, oVbx

   DEFINE DIALOG oDlg FROM 5, 5 TO 20, 40 TITLE "Testing Dialog"


   ACTIVATE DIALOG oDlg CENTERED

return nil

//----------------------------------------------------------------------------//
</code>
